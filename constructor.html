物件建構式

一種用於產生物件資料的函數。

        
// 設計一個用於建立產品的建構式
function Product(name, price) {
    this.name = name;
    this.price = price;
    this.createdAt = new Date().getTime();

    this.intro = function () {
        console.log(`產品名稱:${this.name} 產品價格:${this.price}`)
    }
}
// 建立一項產品
const product1 = new Product('紅茶', 10);
// 讓產品使用intro函數
product1.intro();
// 產品名稱:紅茶 產品價格:10
